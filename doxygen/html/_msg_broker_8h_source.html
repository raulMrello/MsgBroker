9  * - Topics: messages exchanged between software agents: publishers and subscribers.

264  static void publish (const char * topic_name, void *data, uint32_t datasize, MsgBroker::Exception *exception = 0){
265  THROW(NO_ERRORS);

267  THROW(DEINIT);
268  return;
269  }

272  if(!topic){
273  THROW(TOPIC_NOT_FOUND);

276  }
277  if(topic->count > 0){
278  THROW(TOPIC_PENDING);
279  MUTEX_UNLOCK();
280  return;
281  }
282  if(datasize != topic->datasize){
283  THROW(TOPIC_SIZE);
284  MUTEX_UNLOCK();

286  }
287  topic->data = data;

290  while(obs){
291  topic->count++;
292  obs->notify(obs->observer, topic->name);
293  obs = topic->observer_list->getNextItem();
294  }
295  MUTEX_UNLOCK();
